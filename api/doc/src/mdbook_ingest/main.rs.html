1//! # mdbook-ingest (The Ingestion Layer)
2//!
3//! A professional, modular asset ingestion bridge for `mdbook`. This crate serves 
4//! as the **Ingestion Layer** within an autonomous **Research-to-Publish Workflow**.
5//!
6//! ## Agentic Architecture
7//! `mdbook-ingest` is designed to operate as a standalone module that can be 
8//! triggered by a Research Agent. It handles:
9//! 1. **Sanitization**: Transforming raw AI output into hardened Markdown.
10//! 2. **Media Management**: Migrating and renaming assets based on a **Master Key** (Episode Number).
11//! 3. **Indexing**: Maintaining the book's structure and social/monetization widgets.
12//!
13//! Currently, the tool includes specialized logic for **Google Gemini Pro** 
14//! (Shielded Output stripping), with future support planned for other LLMs 
15//! like ChatGPT and Claude.
16

27#[derive(Deserialize, Default)]
28struct BookToml {
29    preprocessor: Option<PreprocessorSection>,
30}

32#[derive(Deserialize, Default)]
33struct PreprocessorSection {
34    ingest: Option<IngestConfig>,
35}

37/// Configuration options loaded from `book.toml` under `[preprocessor.ingest]`
38#[derive(Deserialize, Default, Clone)]
39pub struct IngestConfig {
40    pub downloads_path: Option<String>,
41    pub lightning_address: Option<String>,
42    pub podcast_html: Option<String>,
43    pub title_word_limit: Option<usize>,
44}
45
46impl IngestConfig {
47    pub fn load() -> Self {
48        if let Ok(content) = std::fs::read_to_string("book.toml") {
49            if let Ok(toml) = toml::from_str::<BookToml>(&content) {
50                if let Some(prep) = toml.preprocessor {
51                    if let Some(ingest) = prep.ingest {
52                        return ingest;
53                    }
54                }
55            }
56        }
57        Self::default()
58    }
59}

103fn main() -> Result<()> {
104    // 1. Run "Doctor" checks implicitly on every run (except doctor itself)
105    let cli = Cli::parse();
106    let config = IngestConfig::load();
107    
108    match &cli.command {
109        Some(Commands::Doctor) => {
110            return run_doctor();
111        }
112        _ => {
113            // Implicit check: only warn, don't block yet, to allow installation
114            if let Err(e) = check_dependencies() {
115                eprintln!("⚠️  Dependency Warning: {}", e);
116            }
117        }
118    }
119
120    // 2. Handle Subcommands
121    if let Some(command) = cli.command {
122        match command {
123            Commands::Supports { renderer } => {
124                if renderer == "html" {
125                    std::process::exit(0);
126                } else {
127                    std::process::exit(1);
128                }
129            }
130            _ => {}
131        }
132    }
133
134    let mut source = cli.source.clone();
135    
136    if let Some(config_path) = &config.downloads_path {
137        if cli.source == "/mnt/c/Users/ashut/Downloads" {
138            source = config_path.clone();
139            eprintln!("📂 Using source from book.toml: {}", source);
140        } else {
141            eprintln!("💡 Overriding book.toml source with CLI flag: {}", source);
142        }
143    } else if cli.source != "/mnt/c/Users/ashut/Downloads" {
144        eprintln!("📂 Using source from CLI flag: {}", source);
145    }
146
147    // 3. Handle Ingestion
148    if cli.text {
149        if let Some(number) = cli.number {
150            ingest_text(&number, &source, cli.title.as_deref(), &config)?;
151        } else {
152            anyhow::bail!("❌ Error: Episode number (-n, --number) is required.");
153        }
154    } else if cli.image {
155        if let Some(number) = cli.number {
156            ingest_image(&number, &source, &config)?;
157        } else {
158            anyhow::bail!("❌ Error: Episode number (-n, --number) is required.");
159        }
160    } else if cli.video {
161        if let Some(number) = cli.number {
162            ingest_video(&number, &source, &config)?;
163        } else {
164            anyhow::bail!("❌ Error: Episode number (-n, --number) is required.");
165        }
166    } else if cli.number.is_some() {
167        eprintln!("ℹ️  Please specify asset type (e.g., --text or --image)");
168    } else {
169        // Standard mdbook preprocessor handshake
170        let (_ctx, book) = mdbook::preprocess::CmdPreprocessor::parse_input(std::io::stdin())?;
171        let json = serde_json::to_string(&book)?;
172        print!("{}", json);
173    }
174
175    Ok(())
176}

180    eprintln!("🎬 Ingesting video for episode {}...", number);
181    
182    // 1. Discovery & Migration
183    let vid_dir = "src/vid";
184    std::fs::create_dir_all(vid_dir)?;
185
186    let epi_regex = Regex::new(r"^\d{3}-").unwrap();
187
188    let pattern = format!("{}/*{}*.mp4", source, number);
189    if let Ok(paths) = glob(&pattern) {
190        for path in paths.filter_map(Result::ok) {
191            let filename = path.file_name().unwrap().to_str().unwrap();
192            // Strictly enforce ###- pattern for migration
193            if epi_regex.is_match(filename) {
194                let dest = format!("{}/{}", vid_dir, filename);
195                std::fs::copy(&path, &dest)?;
196                eprintln!("✅ Migrated: {}", filename);

415/// Orchestrates the ingestion of text-based research assets.
416fn ingest_text(number: &str, source: &str, title: Option<&str>, config: &IngestConfig) -> Result<()> {
417    eprintln!("📖 Ingesting text for episode {}...", number);
418    
419    // Find the latest .md, .zip, or .rs file
420    let md_pattern = format!("{}/*.md", source);
421    let zip_pattern = format!("{}/*.zip", source);
422    let rs_pattern = format!("{}/*.rs", source);
423    
424    let mut md_files: Vec<PathBuf> = glob(&md_pattern)?.filter_map(Result::ok)
425        .filter(|p| {
426            let filename = p.file_name().unwrap().to_str().unwrap();
427            !["SUMMARY.md", "cover.md", "README.md"].contains(&filename)
428        }).collect();
429        
430    let mut zip_files: Vec<PathBuf> = glob(&zip_pattern)?.filter_map(Result::ok).collect();
431    let mut rs_files: Vec<PathBuf> = glob(&rs_pattern)?.filter_map(Result::ok).collect();
432    
433    let mut all_files = Vec::new();
434    all_files.append(&mut md_files);
435    all_files.append(&mut zip_files);
436    all_files.append(&mut rs_files);
437
438    all_files.sort_by(|a, b| {
439        let metadata_a = std::fs::metadata(a).unwrap();
440        let metadata_b = std::fs::metadata(b).unwrap();
441        metadata_b.modified().unwrap().cmp(&metadata_a.modified().unwrap())
442    });
443
444    let latest_file = all_files.first().context(format!("❌ No .md, .zip, or .rs files found in {}", source))?;
445    eprintln!("📄 Found export: {:?}", latest_file.file_name().unwrap());
446
447    let content = if latest_file.extension().unwrap() == "zip" {
448        extract_html_from_zip(latest_file)?
449    } else {
450        std::fs::read_to_string(latest_file)?
451    };
452
453    let processed = sanitizer::process_content(content, number, title, config.title_word_limit.unwrap_or(5));
454
455    let dest_path = format!("src/{}.md", number);
456    std::fs::write(&dest_path, processed)?;
457    eprintln!("✅ Saved text to: {}", dest_path);
458
459    update_summary(number, &dest_path)?;
460    
461    Ok(())
462}

464/// Legacy/Fallback: Extracts HTML from ZIP.
465fn extract_html_from_zip(zip_path: &PathBuf) -> Result<String> {
466    let file = std::fs::File::open(zip_path)?;
467    let mut archive = zip::ZipArchive::new(file)?;
468    
469    for i in 0..archive.len() {
470        let mut file = archive.by_index(i)?;
471        if file.name().ends_with(".html") {
472            let mut html_content = String::new();
473            use std::io::Read;
474            file.read_to_string(&mut html_content)?;
475            
476            let eq_regex = regex::Regex::new(r#"<img[^>]*class="[^"]*Math[^"]*"[^>]*title="([^"]*)"[^>]*>"#).unwrap();
477            let preprocessed_html = eq_regex.replace_all(&html_content, |caps: &regex::Captures| {
478                format!(" ${}$ ", caps.get(1).unwrap().as_str())
479            }).to_string();
480
481            return Ok(html2md::parse_html(&preprocessed_html));
482        }
483    }
484    anyhow::bail!("❌ No HTML file found in ZIP archive")
485}

612/// Updates the `SUMMARY.md` file to include the newly ingested episode.
613fn update_summary(number: &str, file_path: &str) -> Result<()> {
614    let summary_path = "src/SUMMARY.md";
615    if !std::path::Path::new(summary_path).exists() { return Ok(()); }
616
617    let mut summary_content = std::fs::read_to_string(summary_path)?;
618    let content = std::fs::read_to_string(file_path)?;
619    let title_line = content.lines()
620        .find(|line| line.starts_with("# "))
621        .map(|line| line.trim_start_matches("# ").trim())
622        .unwrap_or("Untitled");
623
624    let mut title = title_line.replace("**", "").trim().to_string();
625    if let Some(pos) = title.find(':') {
626        title = title[pos + 1..].trim().to_string();
627    }
628
629    let new_entry = format!("- [{} : {}]({}.md)", number, title, number);
630    let start_marker = "<!-- RECENT_START -->";
631    let end_marker = "<!-- RECENT_END -->";
632
633    if let (Some(start_idx), Some(end_idx)) = (summary_content.find(start_marker), summary_content.find(end_marker)) {
634        let insert_pos = start_idx + start_marker.len();
635        let recent_section = &summary_content[insert_pos..end_idx];
636        let mut entries: Vec<String> = recent_section.trim().lines().map(|l| l.to_string()).filter(|l| !l.is_empty()).collect();
637        entries.retain(|l| !l.contains(&format!("({}.md)", number)));
638        entries.insert(0, new_entry);
639        let new_recent_content = format!("\n{}\n", entries.join("\n"));
640        summary_content.replace_range(insert_pos..end_idx, &new_recent_content);
641        std::fs::write(summary_path, summary_content)?;
642        eprintln!("📑 Updated SUMMARY.md");
643    }
644
645    Ok(())
646}
