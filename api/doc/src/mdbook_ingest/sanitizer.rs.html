1//! # Sanitizer Module (Gemini-to-mdbook)
2//! 
3//! This module provides the core transformation logic for converting 
4//! "shielded" Gemini Pro outputs into production-ready mdbook content.
5//! 
6//! It is specifically designed to handle the constraints of the **Master Prompt**, 
7//! including the removal of Rust-style raw string literals (`r#" ... "#`) and 
8//! the enforcement of research-tier formatting.
9
10use regex::Regex;
11use std::collections::HashMap;
12use html_escape::decode_html_entities;

14/// The primary entry point for the Gemini-to-mdbook transformation.
15/// 
16/// This function executes a multi-stage sanitization pipeline:
17/// 1. Decodes HTML entities (to handle web-pasted content).
18/// 2. **Shield Stripping**: Removes the `r#"` and `"#` wrappers and Gemini backticks.
19/// 3. Title Sync: Enforces a word limit for H1 headers.
20/// 4. Unicode Sanitization: Strips invisible control characters and hidden artifacts.
21/// 5. Structural Wrapping: Detects ASCII tables/diagrams and applies correct Markdown formatting.
22/// 6. Footnote Hardening: Re-indexes and formats bibliography sections for KaTeX compatibility.
23pub fn process_content(mut content: String, ep_num: &str, title_override: Option<&str>, word_limit: usize) -> String {
24    // 0. Decode HTML Entities early
25    content = decode_html_entities(&content).to_string();
26
27    // 0.1 Strip Rust raw string artifacts and Gemini markdown markers (The "Shield")
28    if content.starts_with("r#\"") {
29        content = content[3..].to_string();
30    }
31    if content.ends_with("\"#") {
32        content = content[..content.len()-2].to_string();
33    }
34    // Also handle possible trailing newline
35    if content.ends_with("\"#\n") {
36        content = content[..content.len()-3].to_string();
37    }
38    let gemini_artifacts = Regex::new(r"(?m)^```(text|markdown|rust)?\s*$").unwrap();
39    content = gemini_artifacts.replace_all(&content, "").to_string();
40
41    // 0.2 Strip orphaned base64 images early to prevent interference with diagram wrapping
42    let image_def_regex = Regex::new(r"(?m)^\[image\d+\]: <data:image/.*?>\s*$").unwrap();
43    content = image_def_regex.replace_all(&content, "").to_string();
44
45    // 1. Title Sync - Handle "XXX : Title" format and enforce word limit
46    let h1_regex = Regex::new(r"(?m)^#\s(?:\d+\s*:\s*)?\s*(.*)$").unwrap();
47    
48    let display_title = if let Some(t) = title_override {
49        t.to_string()
50    } else if let Some(caps) = h1_regex.captures(&content) {
51        let raw_title = caps.get(1).unwrap().as_str().trim();
52        let clean_title = raw_title.trim_matches('*').trim();
53        
54        // Enforce word limit based on config
55        let words: Vec<&str> = clean_title.split_whitespace().collect();
56        if words.len() > word_limit {
57            let mut truncated = words[..word_limit].join(" ");
58            let stop_words = vec!["of", "the", "in", "and", "on", "a", "an", "with", "for", "to", "at", "by", "from"];
59            if let Some(last_word) = words[..word_limit].last() {
60                if stop_words.contains(&last_word.to_lowercase().as_str()) {
61                    truncated = words[..word_limit-1].join(" ");
62                }
63            }
64            truncated
65        } else {
66            clean_title.to_string()
67        }
68    } else {
69        "Untitled".to_string()
70    };
71    
72    let new_h1 = format!("# {} : {}", ep_num, display_title);
73    if h1_regex.is_match(&content) {
74        content = h1_regex.replace(&content, new_h1.as_str()).to_string();
75    } else {
76        content = format!("{}\n\n{}", new_h1, content);
77    }
78
79    // 2. Invisible Character Sanitization
80    content = content.replace('\u{0332}', "");
81    let control_chars = Regex::new(r"[\x00-\x08\x0b\x0c\x0e-\x1f\x7f-\x9f]").unwrap();
82    content = control_chars.replace_all(&content, "").to_string();
83
84    // 3. Convert ASCII Tables to Markdown Tables (DO THIS BEFORE DIAGRAMS)
85    content = convert_ascii_tables(content);
86
87    // 4. Wrap ASCII Diagrams in code blocks
88    content = wrap_ascii_diagrams(content);
89
90    // 5. Backslash cleanup (Surgical - only common AI escapes)
91    let backslash_cleanup = Regex::new(r"\\([_.\-+!|>\[\]=])").unwrap();
92    content = backslash_cleanup.replace_all(&content, "$1").to_string();
93
94    // 6. Preserve Math Blocks
95    let math_block_regex = Regex::new(r"(?s)\$\$.*?\$\$").unwrap();
96    let inline_math_regex = Regex::new(r"\$.*?\$").unwrap();
97    let mut math_blocks = Vec::new();
98    let content_with_placeholders = math_block_regex.replace_all(&content, |caps: &regex::Captures| {
99        let placeholder = format!("__MATH_BLOCK_{}__", math_blocks.len());
100        math_blocks.push(caps.get(0).unwrap().as_str().to_string());
101        placeholder
102    }).to_string();
103    let content_with_all_placeholders = inline_math_regex.replace_all(&content_with_placeholders, |caps: &regex::Captures| {
104        let placeholder = format!("__MATH_BLOCK_{}__", math_blocks.len());
105        math_blocks.push(caps.get(0).unwrap().as_str().to_string());
106        placeholder
107    }).to_string();
108
109    let mut temp_content = content_with_all_placeholders;
110
111    // 7. Fix Footnotes
112    temp_content = fix_footnotes(temp_content);
113
114    // 8. Currency and Dollar Escaping
115    let currency_regex = Regex::new(r"(?i)\$([\d\.,]+)\s*(million|billion|trillion|k|m|b|t)?").unwrap();
116    temp_content = currency_regex.replace_all(&temp_content, "$1 $2 USD ").to_string();
117    temp_content = temp_content.replace("  ", " ");
118    temp_content = temp_content.replace("$", r"\$");
119
120    // 9. Clean any remaining image tags if they exist
121    let image_tag_regex = Regex::new(r"!\[\]\[image\d+\]").unwrap();
122    temp_content = image_tag_regex.replace_all(&temp_content, "").to_string();
123
124    // 10. Restore Math Blocks
125    for (idx, block) in math_blocks.iter().enumerate() {
126        let placeholder = format!("__MATH_BLOCK_{}__", idx);
127        temp_content = temp_content.replace(&placeholder, block);
128    }
129
130    temp_content.trim().to_string()
131}

133/// Identifies and wraps ASCII-based diagrams in code blocks.
134/// 
135/// Logic detects common diagram markers like `===>`, `|`, and `v` while 
136/// ignoring standard citations or Markdown lists.
137fn wrap_ascii_diagrams(content: String) -> String {
138    let mut result = String::new();
139    let lines: Vec<&str> = content.lines().collect();
140    let mut i = 0;
141    
142    let is_diag_line = |s: &str| -> bool {
143        let t = s.trim();
144        if t.is_empty() { return false; }
145        if t == "v" || t == "|" || t == "^" { return true; }
146        if t.starts_with("v ") || t.starts_with("| ") { return true; }
147        if t.contains("===>") || t.contains("<===") { return true; }
148        if t.starts_with('[') && t.contains(']') && !t.contains("](") { 
149            let citation_regex = Regex::new(r"^\[\d+\]").unwrap();
150            if citation_regex.is_match(t) { return false; }
151            return true; 
152        }
153        if t.starts_with("**<---") || t.starts_with("--->**") { return true; }
154        if t.starts_with("**") && (t.ends_with("v**") || t.ends_with("|**")) { return true; }
155        false
156    };
157
158    while i < lines.len() {
159        if is_diag_line(lines[i]) {
160            let mut last_diag_idx = i;
161            let mut j = i + 1;
162            
163            while j < lines.len() {
164                let t = lines[j].trim();
165                if is_diag_line(lines[j]) {
166                    last_diag_idx = j;
167                } else if !t.is_empty() {
168                    let is_indented = lines[j].starts_with("  ");
169                    let mut next_is_diag = false;
170                    for k in 1..=3 {
171                        if j + k < lines.len() && is_diag_line(lines[j + k]) {
172                            next_is_diag = true;
173                            break;
174                        }
175                    }
176                    if !is_indented || !next_is_diag {
177                        break;
178                    }
179                }
180                j += 1;
181            }
182            
183            let is_multi_line = last_diag_idx > i;
184            let contains_arrows = lines[i].contains("===>") || lines[i].contains("<===");
185            let has_table_separator = (i..=last_diag_idx).any(|k| lines[k].contains("--- |") || lines[k].contains("| ---"));
186            
187            if (is_multi_line || contains_arrows) && !has_table_separator {
188                result.push_str("\n```text\n");
189                for k in i..=last_diag_idx {
190                    result.push_str(lines[k]);
191                    result.push('\n');
192                }
193                result.push_str("```\n");
194            } else {
195                for k in i..=last_diag_idx {
196                    result.push_str(lines[k]);
197                    result.push('\n');
198                }
199            }
200            
201            i = last_diag_idx + 1;
202        } else {
203            result.push_str(lines[i]);
204            result.push('\n');
205            i += 1;
206        }
207    }
208    result
209}

211/// Standardizes and re-indexes footnotes.
212/// 
213/// This function:
214/// - Extracts the "Works cited" or "References" section.
215/// - Re-numbers inline citations `[n]` to `[^n]`.
216/// - Re-numbers the reference list sequentially.
217/// - Ensures URLs are properly hyperlinked in the bibliography.
218fn fix_footnotes(content: String) -> String {
219    let header_regex = Regex::new(r"(?i)#### \*\*Works cited\*\*|#### \*\*References\*\*|## Bibliography|## References or Bibliography").unwrap();
220    let parts: Vec<&str> = header_regex.split(&content).collect();
221    if parts.len() < 2 { return content; }
222
223    let body = parts[0];
224    let refs_raw = parts[1];
225    let header = "#### **Works cited**";
226
227    let ref_pattern = Regex::new(r"(?m)^\*?\s*(\*\*\[?(\d+)\]?\*\*|\*\*\*\*|\[(\d+)\])\s*").unwrap();
228    
229    let mut ref_entries = Vec::new();
230    let matches: Vec<_> = ref_pattern.find_iter(refs_raw).collect();
231    
232    for (i, m) in matches.iter().enumerate() {
233        let caps = ref_pattern.captures(m.as_str()).unwrap();
234        let old_num = caps.get(2).map(|n| n.as_str().to_string())
235            .or_else(|| caps.get(3).map(|n| n.as_str().to_string()));
236        
237        let start = m.end();
238        let end = if i + 1 < matches.len() {
239            matches[i+1].start()
240        } else {
241            refs_raw.len()
242        };
243        
244        let text = refs_raw[start..end].trim().to_string();
245        ref_entries.push(RefEntry { old_num, text, processed: false });
246    }
247
248    if ref_entries.is_empty() { return content; }
249
250    let marker_pattern = Regex::new(r"\[(\d+(?:\s*,\s*\d+)*)\]").unwrap();
251    let mut unique_old_nums = Vec::new();
252    
253    for caps in marker_pattern.captures_iter(body) {
254        let nums_str = caps.get(1).unwrap().as_str();
255        for n in nums_str.split(',') {
256            let n_trimmed = n.trim().to_string();
257            if !unique_old_nums.contains(&n_trimmed) {
258                unique_old_nums.push(n_trimmed);
259            }
260        }
261    }
262
263    let mut old_to_new = HashMap::new();
264    let mut new_refs = Vec::new();
265    let url_regex = Regex::new(r"(https?://[^\s)\]]+[^.\s)\]])").unwrap();
266    
267    for old_num in unique_old_nums {
268        let mut aggregated_text = String::new();
269        let mut found = false;
270        
271        while let Some(ref_entry) = ref_entries.iter_mut().find(|r| r.old_num.as_ref() == Some(&old_num) && !r.processed) {
272            if found { aggregated_text.push_str(" | "); }
273            
274            let entry_text_clean = ref_entry.text.replace("`", "");
275            let processed_text = url_regex.replace_all(&entry_text_clean, |caps: &regex::Captures| {
276                let url = caps.get(1).unwrap().as_str();
277                format!("[{}]({})", url, url)
278            }).to_string();
279            
280            aggregated_text.push_str(&processed_text);
281            ref_entry.processed = true;
282            found = true;
283        }
284
285        if found {
286            let new_num = (new_refs.len() + 1).to_string();
287            old_to_new.insert(old_num, new_num);
288            new_refs.push(aggregated_text);
289        } else if let Some(unprocessed_star) = ref_entries.iter_mut().find(|r| r.old_num.is_none() && !r.processed) {
290            let new_num = (new_refs.len() + 1).to_string();
291            old_to_new.insert(old_num, new_num);
292            
293            let processed_text = url_regex.replace_all(&unprocessed_star.text, |caps: &regex::Captures| {
294                let url = caps.get(1).unwrap().as_str();
295                format!("[{}]({})", url, url)
296            }).to_string();
297            
298            new_refs.push(processed_text);
299            unprocessed_star.processed = true;
300        } else {
301            let new_num = (new_refs.len() + 1).to_string();
302            old_to_new.insert(old_num.clone(), new_num);
303            new_refs.push(format!("**TODO: Missing citation for index {}**", old_num));
304        }
305    }
306
307    let final_body = marker_pattern.replace_all(body, |caps: &regex::Captures| {
308        let nums_str = caps.get(1).unwrap().as_str();
309        let mut new_markers: Vec<String> = nums_str.split(',')
310            .map(|n| {
311                let n_trimmed = n.trim();
312                format!("[^{}]", old_to_new.get(n_trimmed).unwrap_or(&n_trimmed.to_string()))
313            }).collect();
314        new_markers.dedup();
315        new_markers.join(" ") 
316    });
317
318    let mut result = final_body.to_string();
319    result.push_str("\n\n");
320    result.push_str(header);
321    result.push_str("\n\n");
322    
323    for (i, text) in new_refs.iter().enumerate() {
324        result.push_str(&format!("[^{}]: {}\n\n", i + 1, text));
325    }
326    result
327}

331    text: String,
332    processed: bool,
333}

335/// Identifies and converts ASCII grid tables to standard Markdown tables.
336fn convert_ascii_tables(content: String) -> String {
337    let mut result = String::new();
338    let mut lines = content.lines().peekable();
339
340    while let Some(line) = lines.next() {
341        let trimmed = line.trim();
342        let is_table_start = (trimmed.contains("+---") && trimmed.ends_with('+')) || 
343                             (trimmed.starts_with('|') && trimmed.ends_with('|')) ||
344                             (trimmed.starts_with("![][image") && trimmed.contains("+---"));
345        
346        if is_table_start {
347            let mut table_data = Vec::new();
348            table_data.push(line);
349            while let Some(next) = lines.peek() {
350                let nt = next.trim();
351                if nt.starts_with('|') || (nt.contains("+---") && nt.ends_with('+')) || nt.is_empty() {
352                    table_data.push(lines.next().unwrap());
353                } else {
354                    break;
355                }
356            }
357            let mut md_rows = Vec::new();
358            for row in table_data {
359                let mut r = row.trim().to_string();
360                if r.starts_with("![][image") {
361                    if let Some(pos) = r.find('+') { r = r[pos..].to_string(); }
362                    else if let Some(pos) = r.find('|') { r = r[pos..].to_string(); }
363                }
364                if r.starts_with('|') {
365                    let cells: Vec<String> = r.split('|').map(|s| s.trim().to_string()).filter(|s| !s.is_empty()).collect();
366                    if !cells.is_empty() { md_rows.push(cells); }
367                }
368            }
369            if !md_rows.is_empty() {
370                result.push('\n');
371                let mut start_idx = 0;
372                if md_rows[0].len() == 1 {
373                    result.push_str(&format!("**{}**\n\n", md_rows[0][0]));
374                    start_idx = 1;
375                }
376                if md_rows.len() > start_idx {
377                    for (i, row) in md_rows[start_idx..].iter().enumerate() {
378                        result.push_str(&format!("| {} |\n", row.join(" | ")));
379                        if i == 0 {
380                            let sep: Vec<String> = row.iter().map(|_| "---".to_string()).collect();
381                            result.push_str(&format!("| {} |\n", sep.join(" | ")));
382                        }
383                    }
384                }
385                result.push('\n');
386            }
387        } else {
388            result.push_str(line);
389            result.push('\n');
390        }
391    }
392    result
